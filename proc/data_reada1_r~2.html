subroutine data_readA1_R(pname,Y1)
  integer               :: i,Np
  character(len=*)      :: pname
  real(8),dimension(:)  :: Y1
  !
  call file_gunzip(reg(pname))
  call ioread_control(pname,control)
  !
  open(free_unit(unit),file=reg(pname))
  !
  Np=size(Y1)
  !
  do i=1,Np
     read(unit,*)Y1(i)
  enddo
  close(unit)
  call file_gzip(reg(pname))
end subroutine data_readA1_R
