  function fft_farray(L,dt,df) result(freq)
    integer              :: L
    real(8)              :: dt
    real(8),optional     :: df
    real(8)              :: wmax,dw
    real(8),dimension(L) :: freq
    wmax = fft_fmax(L,dt)
    freq = linspace(-wmax,wmax,L,iend=.false.,mesh=dw)
    if(present(df))df=dw
  end function fft_farray
