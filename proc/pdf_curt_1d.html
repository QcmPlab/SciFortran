function pdf_curt_1d(self) result(curt)
  type(pdf_kernel) :: self
  real(8)          :: mean,var,curt
  mean = pdf_mean_1d(self)
  var  = pdf_var_1d(self)
  curt = pdf_moment_1d(self,4,mean)
  curt = curt/var**2
end function pdf_curt_1d
