function pdf_var_1d(self) result(var)
  type(pdf_kernel) :: self
  real(8)          :: var,mean
  mean = pdf_mean_1d(self)
  var  = simps((self%x-mean)**2*self%pdf,self%xmin,self%xmax)
end function pdf_var_1d
