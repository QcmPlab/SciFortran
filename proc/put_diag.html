  subroutine put_diag(diagv,mat)
    real(8), dimension(:), intent(in)      :: diagv
    real(8), dimension(:,:), intent(inout) :: mat
    integer                                :: j,n
    n=assert_eq2(size(diagv),min(size(mat,1),size(mat,2)),'put_diag_rv')
    do j=1,n
       mat(j,j)=diagv(j)
    end do
  end subroutine put_diag
