  subroutine save_input_file(file)
    character(len=*)   :: file
    if(IOinput)then
       call print_input_list(trim(file))
    else
       call print_input_list(trim(file))
       write(*,*)"input file can not be found. Dumped a *default* version in: used."//trim(file)
       stop
    endif
  end subroutine save_input_file
